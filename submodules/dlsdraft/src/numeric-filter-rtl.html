<script>

    // IMPORTANT: the slider will change to RTL, by adding direction: 'rtl' in the define properties

    // pick your slider element
    var zeSlider = document.querySelector('#zeSlider');

    // define the props of it
    var sliderProps = {
        start: [ 500,5000 ],
        direction: 'rtl', // ! In a RTL scenario, specify direction as a property!
        connect: true,
        range: {
            'min': 0,
            'max': 7000
        },
        labelPre:'$',
        labelPost:''
    }

    // you shouldn't be editing past here, but feel free to do it if you want.

    noUiSlider.create(zeSlider, sliderProps);

    var rangeTagLower = document.createElement('div');
    var rangeTagHigher = document.createElement('div');
    rangeTagLower.classList.add('cmx-num-filter__range-label');
    rangeTagHigher.classList.add('cmx-num-filter__range-label');
    zeSlider.querySelector('.noUi-handle-lower').appendChild(rangeTagLower);
    zeSlider.querySelector('.noUi-handle-upper').appendChild(rangeTagHigher);

    zeSlider.noUiSlider.on('update', function( values, handle ) {
        var value = values[handle];
        if ( handle ) {
            rangeTagHigher.innerHTML = sliderProps.labelPre + Math.round(value) + sliderProps.labelPost;
        } else {
            rangeTagLower.innerHTML = sliderProps.labelPre + Math.round(value) + sliderProps.labelPost;
        }
    });
</script>
