                if ((/Chrome/.test(navigator.userAgent) && /Google Inc/.test(navigator.vendor))||(/Safari/.test(navigator.userAgent) && /Apple Computer/.test(navigator.vendor))){
                    var slider = document.getElementById('pictureSizeSlider');
                    var updateSliderTrack = function(e,tgt){
                        var t;
                        if (typeof(tgt)!=='undefined'){
                            t = tgt;
                        } else {
                            t = e.target;
                        }
                        

                        var minVal = t.min;
                        var maxVal = t.max;
                        var val = t.value;

                        var pc = ((val-minVal)*100) / (maxVal-minVal);
                        pc = Math.ceil(pc);

                        t.setAttribute('class','cmx-slider cmx-slider--pc-' + pc);
                    };
                    updateSliderTrack(null,slider);
                    slider.addEventListener('input',updateSliderTrack);
                    slider.addEventListener('change',updateSliderTrack);
                }
            </script>
